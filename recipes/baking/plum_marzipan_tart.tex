\title{Plum Tart}
\maketitle
\subtitle{an hour and a half plus cooling time}

\noindent \desc{Around my school there are two plum trees. In early September, they often produce fruit and with friends I make plum tart. This was adapted from a 1970s era newspaper clipping, adding in more nuts and oats for a nicer texture and richer taste, and cutting back a little bit of sugar. It ends up with a very nice plum tart.}

\recipe{
\textbf{Almond Crust:}
\begin{ingredients}
  \item $\frac{3}{4}$ cup all purpose flour
  \item $\frac{3}{4}$ cup almond flour
  \item $\frac{3}{4}$ cup almonds 
  \item $\frac{1}{4}$ cup sugar
  \item $\frac{1}{8}$ tsp salt
  \item $\frac{1}{2}$ cup butter
  \item 1 tsp almond extract
  \item 1 egg
\end{ingredients}
\textbf{Crumble:}
\begin{ingredients}
  \item $\frac{1}{2}$ cup all purpose flour
  \item $\frac{1}{2}$ cup almond paste
  \item $\frac{1}{4}$ cup brown sugar
  \item 6 tbsp ($\frac{3}{4}$ stick) butter
  \item $\frac{1}{4}$ cup sliced almonds
  \item $\frac{1}{4}$ cup rolled oats
\end{ingredients}
\textbf{Filling:}
\begin{ingredients}
  \item 1000-1100g plums
  \item $\frac{1}{2}$ cup sugar
  \item 2 $\frac{1}{2}$ tbsp cornstarch
\end{ingredients}
}{
\begin{directions}
  \item Preheat oven to 400 Fahrenheit. Make the filling. Blend the flours, almonds, sugar, and salt until the almonds are finely ground. Add butter, sliced into $\frac{1}{2}$ inch cubes, and blend until the dough resembles a course meal. Add the remaining ingredients. Press into 11" pan and pierce with a fork.
  \item Bake the crust, pressing with the back of a fork every few minutes if it boils, for a bout 25 minutes. Transfer to a rack and reduce oven temperature to 375.
  \item Meanwhile, make the crumble: mix flour, almond paste, and sugar until well combined. Add butter (again in $\frac{1}{2}$ inch cubes) and mix until crumbs form. Mix in almonds and oats.
  \item Slice plums into thin pieces and remove the pits. Then mix all ingredients for the filling together. Place into the tart pan and generously top with the crumble.
  \item Bake tart for 40 minutes, until it has thick bubbles and the top is golden. Let cool and serve.
\end{directions}
}

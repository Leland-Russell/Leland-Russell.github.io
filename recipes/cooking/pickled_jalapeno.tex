\title{Pickled Jalapenos}
\maketitle
\subtitle{A half hour}

\noindent \desc{Every summer we grow a bunch of jalapeno plants from which we make pickled peppers. A 16 oz jar usually lasts about two weeks, we end up putting them on everything. Truth be told, I don't know if this is the most updated recipe we use, this is just an older version in my texts. Though, if you are reading this, you must be in my git repo, so you should maybe be expected a few slightly unfinished recipes.}

\recipe{
\begin{ingredients}
  \item $\frac{3}{4}$ cup water
  \item $\frac{3}{4}$ cup white vinegar
  \item 1 tbsp salt
  \item $\frac{1}{2}$ head of garlic
  \item 10 large Jalapeno peppers
  \item $\frac{1}{2}$ tsp oregano (?)
  \item $\frac{1}{2}$ tsp peppercorns (?)
  \item 1 tbsp sugar (?)
\end{ingredients}
}{
\begin{directions}
  \item Slice jalapenos into roughly cm thick slices, smash all the garlic cloves.
  \item Put all ingredients in a sufficiently large pot and boil until Jalapeno's are beginning to become soft, a few minutes (?)
  \item Let cool and pour into a jar. Keep in fridge for a few months
\end{directions}
}
